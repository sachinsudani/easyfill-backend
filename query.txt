
create table "user"("id" serial not null primary key,"username" varchar(100) not null unique,"password" varchar(255) not null, "email" varchar(255) not null unique, "contact_no" varchar(10), "dob" date, "created_at" timestamp default now(), "updated_at" timestamp default now());

create table "token"("id" serial not null primary key,"jwt" varchar(255) unique not null, "user_id" int);

create table address("id" serial not null primary key,"street_line_1" varchar(100),"area" varchar(100),"locality" varchar(100),"house_no" varchar(100),"post_office" varchar(100),"state" varchar(100),"district" varchar(100),"sub_district" varchar(100),"city" varchar(100),"pincode" varchar(6));
